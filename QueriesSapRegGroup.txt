PROCEDIMIENTO

ALTER PROCEDURE [dbo].[zSP_SapRegGroup] @tynOperacion tinyint, @NameCust varchar(300), @CustID varchar(100), @SAPRG varchar(100)
AS


/* OPERACIONES DE CONSULTA */
IF (@tynOperacion = 1)
BEGIN
	SELECT [Name], C.CustId, ClassId, Country, TaxLocId 
	FROM [AUMXAPPTEST].[dbo].[Customer] C WITH(NOLOCK)
	WHERE ClassID NOT IN ('V', 'DUMMY')
	AND C.CustId NOT IN (SELECT CS.CustID FROM [AUMXAPPTEST].[dbo].[zCustomerSapRegGroup] CS)
END

IF (@tynOperacion = 2)
BEGIN
	SELECT [RegCustGroupID], [Description], [InsertDate], [LastUpdate], [UserCreation] 
	FROM [AUMXAPPTEST].[dbo].[zSapRegGroup] WITH(NOLOCK)
END

IF (@tynOperacion = 3)
BEGIN
	SELECT [Name], CustId, ClassId, Country, TaxLocId 
	FROM [AUMXAPPTEST].[dbo].[Customer] C WITH(NOLOCK)
	WHERE ClassID NOT IN ('V', 'DUMMY') AND [CustId] LIKE @CustID
	AND C.CustId NOT IN (SELECT CS.CustID FROM [AUMXAPPTEST].[dbo].[zCustomerSapRegGroup] CS)
END

IF (@tynOperacion = 4)
BEGIN
	SELECT [Name], CustId, ClassId, Country, TaxLocId 
	FROM [AUMXAPPTEST].[dbo].[Customer] C WITH(NOLOCK)
	WHERE ClassID NOT IN ('V', 'DUMMY') AND [Name] LIKE @NameCust
	AND C.CustId NOT IN (SELECT CS.CustID FROM [AUMXAPPTEST].[dbo].[zCustomerSapRegGroup] CS)
END

IF (@tynOperacion = 5)
BEGIN
	SELECT CSRG.CustId, SapRegGroup, InsertDate, LastUpdate, UserCreation 
	FROM [AUMXAPPTEST].[dbo].[Customer] C WITH(NOLOCK)
	INNER JOIN [AUMXAPPTEST].[dbo].[zCustomerSapRegGroup] CSRG ON C.CustId = CSRG.CustID
	WHERE ClassID NOT IN ('V', 'DUMMY')
END

/* OPERACIONES INSERT */

IF (@tynOperacion = 20)
BEGIN
	INSERT INTO [AUMXAPPTEST].[dbo].[zCustomerSapRegGroup] ([CustID], [SapRegGroup], [InsertDate], [LastUpdate], [UserCreation]) 
	VALUES (@CustID, @SAPRG, GETDATE(), GETDATE(), NULL)
END
